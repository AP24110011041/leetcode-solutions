## Solution

```java
class Solution {
    public int[] sumAndMultiply(String s, int[][] q) {
        int MOD = 1000000007;
        ArrayList<Integer> p = new ArrayList<>(), d = new ArrayList<>();
        for (int i = 0; i < s.length(); i++)
            if (s.charAt(i) != '0') {
                p.add(i);
                d.add(s.charAt(i) - '0');
            }

        int n = d.size();
        long[] pow = new long[n + 1], val = new long[n + 1], sum = new long[n + 1];
        pow[0] = 1;
        for (int i = 0; i < n; i++) {
            pow[i + 1] = pow[i] * 10 % MOD;
            val[i + 1] = (val[i] * 10 + d.get(i)) % MOD;
            sum[i + 1] = sum[i] + d.get(i);
        }

        int[] ans = new int[q.length];
        for (int i = 0; i < q.length; i++) {
            int l = lb(p, q[i][0]), r = ub(p, q[i][1]) - 1;
            if (l > r) continue;
            long x = (val[r + 1] - val[l] * pow[r - l + 1] % MOD + MOD) % MOD;
            ans[i] = (int) (x * (sum[r + 1] - sum[l]) % MOD);
        }
        return ans;
    }

    int lb(ArrayList<Integer> a, int x) {
        int l = 0, r = a.size();
        while (l < r) {
            int m = (l + r) / 2;
            if (a.get(m) < x) l = m + 1;
            else r = m;
        }
        return l;
    }

    int ub(ArrayList<Integer> a, int x) {
        int l = 0, r = a.size();
        while (l < r) {
            int m = (l + r) / 2;
            if (a.get(m) <= x) l = m + 1;
            else r = m;
        }
        return l;
    }
}
```
